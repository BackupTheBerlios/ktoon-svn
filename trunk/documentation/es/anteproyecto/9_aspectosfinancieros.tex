\section{Aspectos financieros}
{
	\subsection{Recursos}
	{
		A continuación se listan los recursos necesarios para llevar a cabo este proyecto:
		\begin{itemize}
			\item{Dos estudiantes de ingenieria de sistema para desarrollar el proyecto}
			\item{Un director de tesis, encargado de guiar a el proyecto.}
			\item{Un asesor, encargado de asesorar en temas de animación a los estudiantes.}
			\item{Al menos dos equipos de computo conectados entre sí.}
			\item{Transporte empleado durante el ciclo de desarrollo del proyecto}
			\item{Fotocopias, libros, CD's, lapices, borradores y material de oficina.}
		\end{itemize}
	}
}
